31def getDistanceByHaversine(loc1, loc2): 

32 "Haversine formula - give coordinates as (lat_decimal,lon_decimal) tuples" 

33 

34 lat1, lon1 = loc1 

35 lat2, lon2 = loc2 

36 

37 # convert to radians 

38 lon1 = lon1 * pi / 180.0 

39 lon2 = lon2 * pi / 180.0 

40 lat1 = lat1 * pi / 180.0 

41 lat2 = lat2 * pi / 180.0 

42 

43 # haversine formula 

44 dlon = lon2 - lon1 

45 dlat = lat2 - lat1 

46 a = (sin(dlat/2))**2 + cos(lat1) * cos(lat2) * (sin(dlon/2.0))**2 

47 c = 2.0 * atan2(sqrt(a), sqrt(1.0-a)) 

48 km = earthradius * c 

49 return km

69def getCoordinatesForDistance(originlat, originlon, distance, units="km"): 

70 """return longitude & latitude values that, when added to & subtraced from 

71 origin longitude & latitude, form a cross / 'plus sign' whose ends are 

72 a given distance from the origin""" 

73 

74 degreelength = 0 

75 

76 if units == "km": 

77 degreelength = degreeInKms 

78 elif units == "miles": 

79 degreelength = degreeInMiles 

80 else: 

81 raise Exception("Units must be either 'km' or 'miles'!") 

82 

83 lat = distance / degreelength 

84 lon = distance / (cos(originlat * rad) * degreelength) 

85 

86 return (lat, lon)
